  <script>
    (function(){
      const area = document.getElementById('spaghettiArea');
      const modal = document.getElementById('recipeModal');
      const closeBtn = document.getElementById('closeModalBtn');

      // Initialize modal state
      let modalVisible = false;
      const modalWidth = 420; // design width; actual width is CSS-controlled

      // Helper to clamp value
      const clamp = (v, min, max) => Math.max(min, Math.min(v, max));

      function positionModal(){
        // Compute horizontal center of the image area
        const areaRect = area.getBoundingClientRect();
        const centerX = areaRect.left + areaRect.width / 2;
        // Determine left to center modal on the image center, but keep within viewport
        const w = Math.min(modalWidth, window.innerWidth * 0.92);
        const desiredLeft = centerX - w / 2;
        const left = clamp(desiredLeft, 12, window.innerWidth - w - 12);

        modal.style.width = w + 'px';
        modal.style.left = left + 'px';
        // Keep the modal near the top (already set to top: 60px in CSS)
      }

      function showModal(){
        if (modalVisible) return;
        modalVisible = true;
        modal.setAttribute('aria-hidden','false');
        modal.classList.add('visible');
        positionModal();
      }

      function hideModal(){
        if (!modalVisible) return;
        modalVisible = false;
        modal.setAttribute('aria-hidden','true');
        modal.classList.remove('visible');
      }

      // Event handling: show on hover/focus of the image area
      area.addEventListener('mouseenter', showModal);
      area.addEventListener('focus', showModal);

      // Keep it visible while cursor is over the modal as well
      modal.addEventListener('mouseenter', showModal);
      modal.addEventListener('focus', showModal);

      // Hide when leaving both image and modal (with small grace)
      area.addEventListener('mouseleave', (e) => {
        // If moving to modal, keep open (handled by modal's mouseenter)
        const toElement = e.relatedTarget;
        if (toElement && modal.contains(toElement)) return;
        setTimeout(() => {
          if (!modal.matches(':hover') && !area.matches(':focus')) hideModal();
        }, 60);
      });

      modal.addEventListener('mouseleave', (e) => {
        // If leaving modal but still over image, keep open
        const toElement = e.relatedTarget;
        if (toElement && area.contains(toElement)) return;
        setTimeout(() => {
          if (!area.matches(':hover') && !modal.matches(':focus')) hideModal();
        }, 60);
      });

      // Keyboard: close with Escape
      document.addEventListener('keydown', (ev) => {
        if (ev.key === 'Escape') hideModal();
      });

      // Re-position on window resize or scroll
      window.addEventListener('resize', () => {
        if (modalVisible) positionModal();
      });
      window.addEventListener('scroll', () => {
        if (modalVisible) positionModal();
      });

      // Close button
      closeBtn.addEventListener('click', hideModal);

      // Optional: click on image area to toggle (for accessibility)
      area.addEventListener('click', () => {
        if (modalVisible) hideModal(); else showModal();
      });
    })();
  </script>
